## Selected

<Playground>
  {() => {
    const [selected, setSelected] = useState(false);
    return (
      <Chip
        avatar={{ name: "Yoda", imageUrl: "https://bit.ly/38rd3Em" }}
        label="Yoda"
        selected={selected}
        onClick={() => setSelected(true)}
        {...(selected && {
          dismissAction: {
            ariaLabel: "Remove",
            onRequestDismiss: () => setSelected(false),
          },
        })}
      />
    );
  }}
</Playground>
